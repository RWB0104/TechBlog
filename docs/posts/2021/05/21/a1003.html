		// ÌîºÎ≥¥ÎÇòÏπò ÏàòÏó¥ Ï¥àÍ∏∞Í∞í (N = 0)

			else

	}
	
	/**
	 * ÌîºÎ≥¥ÎÇòÏπò Í∞í Î∞òÌôò Ìï®Ïàò

	private static int fibonacci(int n)

		if (arr[n] == null)
		{

		}
		
		return arr[n];
	}
