	private static int[] bubbleSort()

		int swapCount = 0;

		boolean isSwaped = true;

		int index = array.length - 1;

		while (isSwaped)
		{

			{

		}

	}

	 * 요소의 중복 여부 반환 함수

	 * @param array: [int[]] 배열
	 *
	 * @return [boolean] 중복 여부

	private static boolean isDuplicated(int[] array)
	{
		for (int i = 0; i < array.length; i++)
		{
			for (int j = 0; j < array.length; j++)

				count++;

					return true;
				}
			}
		}
		
		return false;
	}
